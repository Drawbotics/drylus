import { Layout, LayoutPositions } from '@drawbotics/react-drylus';

import Playground, { PropsTable } from '~/components/Playground';


export const Sidebar = ({ width }) => {
  return (
    <div style={{ width: width || 200, height: '100%', background: 'red', padding: 20 }}>Sidebar</div>
  );
}

export const Navbar = () => {
  return (
    <div style={{ height: 60, width: '100%', background: 'blue', padding: 20 }}>Navbar</div>
  );
}

export const BottomBar = () => {
  return (
    <div style={{ height: 60, width: '100%', background: 'pink', padding: 20 }}>Bottom bar</div>
  );
}
