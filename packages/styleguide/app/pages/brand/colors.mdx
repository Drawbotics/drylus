import sv from '@drawbotics/drylus-style-vars';
import { css, cx } from 'emotion';
import { Grid, GridItem, Sizes } from '@drawbotics/react-drylus';


export const styles = {
  palette: css`
    border-radius: ${sv.defaultBorderRadius};
    overflow: hidden;
  `,
  title: css`
    padding: ${sv.paddingSmall};
    color: ${sv.colorPrimaryInverse};
    text-transform: uppercase;
    font-size: 0.9rem;
  `,
  color: css`
    padding: ${sv.paddingSmall};
    display: flex;
    align-items: center;
    justify-content: space-between;
    transition: ${sv.defaultTransition};
    position: relative;
    &::after {
      content: ' ';
      position: absolute;
      bottom: -10px;
      left: 0;
      height: 10px;
      width: 100%;
      background: currentColor;
    }
    &:hover {
      cursor: pointer;
      transform: translateY(-10px);
    }
    > div {
      color: ${sv.colorPrimaryInverse};
    }
  `,
  light: css`
    > div {
      color: ${sv.colorPrimary};
    }
  `,
}


export const Palette = ({
  palette,
}) => {
  const { colors, title, background } = palette;
  return (
    <div className={styles.palette}>
      <div className={styles.title} style={{ background }}>
        {title}
      </div>
      {colors.map((color, i) => (
        <div
          key={i}
          style={{ background: sv[color.name], color: sv[color.name] }}
          className={cx(styles.color, { [styles.light]: color.light })}>
          <div>{color.name}</div>
          <div>{sv[color.name]}</div>
        </div>
      ))}
    </div>
  );
};


export const Colors = ({ palettes }) => {
  return (
    <Grid
      columns={3}
      hGutters={Sizes.DEFAULT}
      vGutters={Sizes.DEFAULT}>
      {palettes.map((p, i) => (
        <Palette key={i} palette={p} />
      ))}
    </Grid>
  );
}
