import sv from '@drawbotics/drylus-style-vars';
import { css, cx } from 'emotion';
import { Grid, GridItem, Sizes, Tooltip, Icon } from '@drawbotics/react-drylus';

import { greys, neutrals, azures } from './palettes';


export const styles = {
  palette: css`
    border-radius: ${sv.defaultBorderRadius};
    overflow: hidden;
  `,
  header: css`
    padding: ${sv.paddingSmall};
    color: ${sv.colorPrimaryInverse};
    display: flex;
    align-items: center;
    justify-content: space-between;
  `,
  title: css`
    text-transform: uppercase;
    font-size: 0.9rem;
  `,
  color: css`
    padding: ${sv.paddingSmall};
    display: flex;
    align-items: center;
    justify-content: space-between;
    transition: ${sv.defaultTransition};
    position: relative;
    &::after {
      content: ' ';
      position: absolute;
      bottom: -10px;
      left: 0;
      height: 10px;
      width: 100%;
      background: currentColor;
    }
    &:hover {
      cursor: pointer;
      transform: translateY(-10px);
    }
    > div {
      color: ${sv.colorPrimaryInverse};
    }
  `,
  light: css`
    > div {
      color: ${sv.colorPrimary};
    }
  `,
}


export const Palette = ({
  palette,
}) => {
  const { colors, title, background, description } = palette;
  return (
    <div className={styles.palette}>
      <div className={styles.header} style={{ background }}>
        <div className={styles.title}>{title}</div>
        <Tooltip message={description}>
          <Icon name="info" />
        </Tooltip>
      </div>
      {colors.map((color, i) => (
        <div
          key={i}
          style={{ background: sv[color.name], color: sv[color.name] }}
          className={cx(styles.color, { [styles.light]: color.light })}>
          <div>{color.name}</div>
          <div>{sv[color.name]}</div>
        </div>
      ))}
    </div>
  );
};


export const Colors = ({ palettes }) => {
  return (
    <Grid
      columns={3}
      hGutters={Sizes.DEFAULT}
      vGutters={Sizes.DEFAULT}>
      {palettes.map((p, i) => (
        <div key={i}>
          <Palette palette={p} />
        </div>
      ))}
    </Grid>
  );
}


# Colors

### Basics

These are used for elements which shouldn't bring much attention to the user, rather used to create separation between layers, or add texture.
