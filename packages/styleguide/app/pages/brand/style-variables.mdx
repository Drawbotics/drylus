import sv from '@drawbotics/style-vars';
import { Table, Panel, PanelBody, PanelSection } from '@drawbotics/react-drylus';


# Style Variables
Here is a comprehensive list of all the constants exposed by the `style-vars` package. These are only for reference, if you're unsure what value to use for a particular style do ask our designers first.

All of the following (except functions) are available as CSS and LESS variables.

export const Constants = () => {
  const data = Object.keys(sv).map((key) => ({
    name: key,
    value: String(sv[key]),
  }));
  return (
    <Panel body={
      <PanelBody noPadding>
        <PanelSection>
          <Table data={data} header={['name', 'value']} />
        </PanelSection>
      </PanelBody>
    } />
  );
};


<Constants />
