import { Input, Select, Button, Icon } from '@drawbotics/react-drylus';
import { css } from 'emotion';

import Playground from '~/components/Playground';
import { ControlledField } from '../utils';


# Input
Usage
```jsx
import { Input } from '@drawbotics/react-drylus';
```

__Example__
<Playground component={Input}>
  <ControlledField style={{ maxWidth: 250 }} component={
    <Input placeholder="Placeholder..." onChange={x=>x} value="" />
  } />
</Playground>


You can also add a prefix and suffix.

export const style = {
  example: css`
    label: noreplace;
    width: 300px;
    > * {
      margin-bottom: 20px;
    }
  `,
};

export const Example = ({ children }) => {
  return (
    <div className={style.example}>
      {children}
    </div>
  );
}

<Playground>
  <Example>
    <Input placeholder="Something..." prefix="$" value="" onChange={x=>x} />
    <Input placeholder="Something..." suffix=".com" value="" onChange={x=>x} />
    <Input placeholder="Something..." prefix={
      <Select value="option" options={[{ label: 'Option', value: 'option' }]} placeholder="Select..." onChange={x=>x} />
    } value="" onChange={x=>x} />
    <Input placeholder="Something else..." suffix={
      <Button leading={<Icon name="copy" />}>Click me</Button>
    } value="" onChange={x=>x} />
  </Example>
</Playground>
