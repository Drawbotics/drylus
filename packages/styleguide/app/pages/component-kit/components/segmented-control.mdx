import { SegmentedControl, SegmentedControlItem, Callout, Spinner, Badge, Category, Icon, Size } from '@drawbotics/react-drylus';

import Playground from '~/components/Playground';
import { ControlledField } from '../utils';


# Segmented Control
Usage
```jsx
import { SegmentedControl, SegmentedControlItem } from '@drawbotics/react-drylus';
```

__Example__
<Playground component={SegmentedControl}>
  <SegmentedControl>
    <SegmentedControlItem active text="First choice" />
    <SegmentedControlItem text="Second choice" />
    <SegmentedControlItem text="Third choice" />
  </SegmentedControl>
</Playground>

#### SegmentedControl Item
<Playground component={SegmentedControlItem}>
  <SegmentedControlItem text="I'm an item" />
</Playground>

You can also give leading and trailing to items, and disable them:
<Playground>
  <SegmentedControl>
    <SegmentedControlItem active text="First option" trailing={<Spinner size={Size.SMALL}  />} />
    <SegmentedControlItem text="Second option" trailing={<Badge value={5} category={Category.DANGER} />} />
    <SegmentedControlItem text="Third option" disabled leading={<Icon name="brand-id" />} />
  </SegmentedControl>
</Playground>

You can also have a number value displayed within each control through `bullet`. Each option can also be `disabled`.

<Playground>
  <ControlledField component={
    <SegmentedControl
      options={[{
        label: 'First choice',

        value: 'second',
        bullet: 23,
        disabled: true,

        bullet: 123,
        loading: true,
      }]} />

</Playground>
