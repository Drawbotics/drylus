import {
  Callout,
  Category,
} from '@drawbotics/react-drylus';

import Playground from '~/components/Playground';


# Callout
Component used to display a message that complements the content of the page. Similar to [`Banner`](/component-kit/components/banner), but used to the give the user more in depth information about what they're seeing on the page, sort of like tips. Cannot be dismissed.
```jsx
import { Callout } from '@drawbotics/react-drylus';
```

__Example__
<Playground component={Callout}>
  <Callout category={Category.SUCCESS}>
    Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod
  </Callout>
</Playground>
