# Collapsible
This component is used when there is a lot of information on the page, and we want to declutter the UI. This component is __controlled__.
```jsx
import { Collapsible } from '@drawbotics/react-drylus';
```

__Example__

export const ControlledField = ({ component: Component }) => {
  const [isOpen, setIsOpen] = useState(false);
  return (
    <div style={{ maxWidth: 400 }}>
      {React.cloneElement(Component, { isOpen, onClick: () => setIsOpen(! isOpen) })}
    </div>
  );
};


<Playground component={Collapsible}>
  <ControlledField component={
    <Collapsible title="Some information">
      <div style={{ background: 'red', height: 150 }} />
    </Collapsible>
  } />
</Playground>
