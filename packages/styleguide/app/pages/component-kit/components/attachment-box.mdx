import { AttachmentBox, Margin, Size } from '@drawbotics/react-drylus';

import Playground from '~/components/Playground';


# Attachment Box
Used to display attachment components.

```jsx
import { AttachmentBox } from '@drawbotics/react-drylus';
```
__Example__

export const attachment = {
  id: '1',
  fileName: 'DummyFile.jpg',
  progress: 1,
  url: 'https://stackoverflow.com/',
};

export const secondAttachment = {
  id: '2',
  fileName: 'SecondDummyFile.jpg',
  progress: 0.6,
};

export const onClickDownload = () => {};
export const onClickClose = () => {};

<Playground component={AttachmentBox}>
  <Margin size={{ bottom: Size.SMALL }}>
    <AttachmentBox fileName={attachment.fileName} progress={attachment.progress} onClickDownload={onClickDownload} onClickClose={onClickClose}/>
  </Margin>
  <AttachmentBox fileName={secondAttachment.fileName} progress={secondAttachment.progress} onClickDownload={onClickDownload} onClickClose={onClickClose} />
</Playground>
