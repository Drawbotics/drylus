import { RadioGroup, InputGroup } from '@drawbotics/react-drylus';

import Playground from '~/components/Playground';
import { ControlledField } from '../utils';


# Radio Group
Usage
```jsx
import { RadioGroup } from '@drawbotics/react-drylus';
```

__Example__
<Playground component={RadioGroup}>
  <ControlledField component={
    <RadioGroup
      options={[{
        label: 'Belgium',
        value: 'BE',
      }, {
        label: 'France',
        value: 'FR',
      }, {
        label: 'Italy',
        value: 'IT',
      }]} />
  } />
</Playground>


You can also disable individual options by adding `disabled` to the options:
<Playground>
  <ControlledField component={
    <RadioGroup
      options={[{
        label: 'Belgium',
        value: 'BE',
        disabled: true,
      }, {
        label: 'France',
        value: 'FR',
      }, {
        label: 'Italy',
        value: 'IT',
      }]} />
  } />
</Playground>

Not specifying an `onChange` function will render the component `read-only`:
<Playground>
  <InputGroup>
    <RadioGroup
      options={[{
        label: 'Belgium',
        value: 'BE',
      }, {
        label: 'France',
        value: 'FR',
      }, {
        label: 'Italy',
        value: 'IT',
      }]} />
    <RadioGroup
      value="BE"
      options={[{
        label: 'Belgium',
        value: 'BE',
      }, {
        label: 'France',
        value: 'FR',
      }, {
        label: 'Italy',
        value: 'IT',
      }]} />
  </InputGroup>
</Playground>
