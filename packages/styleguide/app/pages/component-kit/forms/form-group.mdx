import {
  FormGroup,
  Label,
  Input,
  InputGroup,
  Checkbox,
  Select,
  Margin,
  Size,
  Hint,
  Flex,
  FlexItem,
  FlexJustify,
  Category,
} from '@drawbotics/react-drylus';

import Playground from '~/components/Playground';


# Form Group
This component sets the layout when combining a `Label` with a form element or multiple form elements (see [`InputGroup`](/component-kit/forms/input-group) for more examples).
Usage
```jsx
import { FormGroup } from '@drawbotics/react-drylus';
```

__Example__
<Playground component={FormGroup}>
  <FormGroup
    input={<Input placeholder="Something..." value="" onChange={x=>x} />}
    label={<Label>My input label</Label>} />
</Playground>


### InputGroup example
As mentioned above you can set the `FormGroup` to take multiple components as inputs, in order to "group" inputs into a single label.

<Playground>
  <FormGroup
    input={
      <InputGroup hint="This is a global hint">
        <Checkbox checked>First one</Checkbox>
        <Checkbox>Second one</Checkbox>
        <Checkbox>Third one</Checkbox>
      </InputGroup>
    }
    label={<Label>My group of inputs</Label>} />
</Playground>

And can be horizontal too of course and have a global error or validation:
<Playground>
  <>
    <FormGroup
      horizontal
      input={
        <InputGroup error="This is a global error">
          <Checkbox checked>First one</Checkbox>
          <Checkbox>Second one</Checkbox>
          <Checkbox>Third one</Checkbox>
        </InputGroup>
      }
      label={<Label>My group of checkboxes</Label>} />
    <Margin size={{ bottom: Size.DEFAULT }} />
    <FormGroup
      horizontal
      input={
        <InputGroup valid>
          <Input value="Something" onChange={x=>x} />
          <Input value="Something else" onChange={x=>x} />
        </InputGroup>
      }
      label={<Label>My group of inputs</Label>} />
  </>
</Playground>


### Address example
Sometimes you might need to create complex form groups, for example for an address section. **Note** you will have to set the layout and display of the hint/error yourself when making this kind of form groups, same goes for the error states of the components. See [`Hint`](/component-kit/forms/hint) documentation for that.

<Playground>
  <FormGroup
    input={
      <>
        <Flex justify={FlexJustify.START} wrap>
          <FlexItem flex={3/4}>
            <Margin size={{ right: Size.EXTRA_SMALL, bottom: Size.EXTRA_SMALL }}>
              <Input valid placeholder="Street & Number" value="1 Real Avenue" onChange={x=>x} />
            </Margin>
          </FlexItem>
          <FlexItem flex={1/4}>
            <Margin size={{ bottom: Size.EXTRA_SMALL }}>
              <Input error placeholder="ZIP" onChange={x=>x} value="" />
            </Margin>
          </FlexItem>
          <FlexItem flex={1/2}>
            <Margin size={{ right: Size.EXTRA_SMALL }}>
              <Input valid placeholder="City" onChange={x=>x} value="Brussels" />
            </Margin>
          </FlexItem>
          <FlexItem flex={1/2}>
            <Select error placeholder="Country" options={[{ value: 'IT', label: 'Italy' }, { value: 'FR', label: 'France' }]} onChange={x=>x} />
          </FlexItem>
        </Flex>
        <Hint category={Category.DANGER}>All fields are required</Hint>
      </>
    }
    label={<Label>Address of residence</Label>} />
</Playground>
