import { SegmentedControl } from '@drawbotics/react-drylus';

import Playground from '~/components/Playground';
import { ControlledField } from '../utils';


# Segmented Control
Usage
```jsx
import { SegmentedControl } from '@drawbotics/react-drylus';
```

__Example__
<Playground component={SegmentedControl}>
  <ControlledField component={
    <SegmentedControl
      options={[{
        label: 'First choice',
        value: 'first',
      }, {
        label: 'Second choice',
        value: 'second',
      }, {
        label: 'Third',
        value: 'third',
      }]} />
  } />
</Playground>


You can also have a number value displayed within each control through `bullet`. Each option can also be `disabled`.

<Playground>
  <ControlledField component={
    <SegmentedControl
      options={[{
        label: 'First choice',
        value: 'first',
        bullet: 4,
      }, {
        label: 'Second choice',
        value: 'second',
        bullet: 23,
        disabled: true,
      }, {
        label: 'Third',
        value: 'third',
        bullet: 123,
        loading: true,
      }]} />
  } />
</Playground>
