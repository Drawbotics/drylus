import { Link, TextLink, Text } from '@drawbotics/react-drylus';

import Playground from '~/components/Playground';


# Link
By default a `Link` sets no style except for the `cursor: pointer`. This is because we might have entire components wrapped in a `Link` to provide navigation.

<Playground component={Link}>
  <Link>Hello Im a link</Link>
</Playground>

### Text Link

If you want to use links within text, the component also provides a `TextLink` component that only accepts `string` as children, and sets some default styling:

<Playground component={TextLink}>
  <>
    <Text>This is some normal text and there is a </Text><TextLink>link</TextLink><Text> within it</Text>
  </>
</Playground>
