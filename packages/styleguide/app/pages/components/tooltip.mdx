import {

  TooltipSides,
  Button,
  Categories,
  Dot,
  Flex,
  FlexItem,
  Sizes,
  Margin,
} from '@drawbotics/react-drylus';

import Playground from '~/components/Playground';


# Tooltip
It can be used to wrap any component/element and display a hint when hovered. The component is uncontrolled and uses portals to be rendered on top of the content.
```jsx
import { Tooltip, TooltipSides } from '@drawbotics/react-drylus';
```

__Example__
<Playground component={Tooltip} enums={{ TooltipSides }}>
<Flex>
  <FlexItem>
    <Margin size={{ right: Sizes.LARGE }}>
      <Tooltip message="I am a tooltip">
        <Button category={Categories.BRAND}>Hover me</Button>
      </Tooltip>
    </Margin>
  </FlexItem>
  <FlexItem>
    <Tooltip message="I am another tooltip">
      <Dot category={Categories.SUCCESS} />
    </Tooltip>
  </FlexItem>
</Flex>
</Playground>
